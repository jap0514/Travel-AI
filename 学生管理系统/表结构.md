#### 1、用户表

CREATE TABLE sys_user (
  id BIGINT NOT NULL AUTO_INCREMENT,
  username VARCHAR(50) NOT NULL UNIQUE COMMENT '用户名/学号/工号',
  password VARCHAR(100) NOT NULL COMMENT '加密密码',
  email VARCHAR(100) COMMENT '邮箱',
  phone VARCHAR(20) COMMENT '手机号',
  avatar VARCHAR(200) COMMENT '头像URL',
  status TINYINT DEFAULT 1 COMMENT '状态：0-禁用 1-正常',
  last_login_time datetime COMMENT '最后登录时间',
  last_login_ip VARCHAR(50) COMMENT '最后登录IP',
  created_time datetime DEFAULT CURRENT_TIMESTAMP,
  updated_time datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  INDEX idx_username (username),
  INDEX idx_email (email)
) COMMENT = '系统用户表';



#### 2、角色表

#### 3、用户角色关联表

CREATE TABLE `sys_user_role` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `user_id` BIGINT NOT NULL,
  `role_id` BIGINT NOT NULL,
  `created_time` datetime DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_user_role` (`user_id`, `role_id`),
  FOREIGN KEY (`user_id`) REFERENCES `sys_user` (`id`) ON DELETE CASCADE,
  FOREIGN KEY (`role_id`) REFERENCES `sys_role` (`id`) ON DELETE CASCADE
) COMMENT = '用户角色关联表';





#### 4、菜单权限表

CREATE TABLE `sys_menu` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `parent_id` BIGINT DEFAULT 0 COMMENT '父菜单ID',
  `menu_name` VARCHAR(50) NOT NULL COMMENT '菜单名称',

  `path` VARCHAR(200) COMMENT '路由路径(后端路径)',
  `component` VARCHAR(200) COMMENT '组件路径(前端路径)',
  `perms` VARCHAR(100) COMMENT '权限标识',
  `icon` VARCHAR(50) COMMENT '图标',
  `sort_order` INT DEFAULT 0 COMMENT '排序',
  `visible` TINYINT DEFAULT 1 COMMENT '是否显示：0-隐藏 1-显示',
  `created_time` datetime DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`)
) COMMENT = '菜单权限表';





#### 5、角色菜单关联表

CREATE TABLE `sys_role_menu` (

#### 7、专业表

#### 9、学生表

CREATE TABLE `student` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `user_id` BIGINT NOT NULL UNIQUE COMMENT '关联用户ID',
  `student_number` VARCHAR(50) NOT NULL UNIQUE COMMENT '学号',
  `name` VARCHAR(50) NOT NULL COMMENT '姓名',
  `gender` TINYINT DEFAULT 0 COMMENT '性别：0-男 1-女',
  `id_card` VARCHAR(18) COMMENT '身份证号',
  `birth_date` date COMMENT '出生日期',
  `nationality` VARCHAR(50) DEFAULT '汉族' COMMENT '民族',
  `political_status` VARCHAR(50) COMMENT '政治面貌',
  `class_id` BIGINT NOT NULL COMMENT '所属班级',
  `enrollment_date` date COMMENT '入学日期',
  `graduation_date` date COMMENT '预计毕业日期',
  `student_status` TINYINT DEFAULT 1 COMMENT '学籍状态：1-在读 2-休学 3-退学 4-毕业',
  `phone` VARCHAR(20) COMMENT '联系电话',
  `email` VARCHAR(100) COMMENT '邮箱',
  `home_address` VARCHAR(200) COMMENT '家庭地址',
  `emergency_contact` VARCHAR(50) COMMENT '紧急联系人',
  `emergency_phone` VARCHAR(20) COMMENT '紧急联系电话',
  `created_time` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  FOREIGN KEY (`user_id`) REFERENCES `sys_user` (`id`),
  FOREIGN KEY (`class_id`) REFERENCES `class` (`id`),
  INDEX `idx_student_number` (`student_number`),
  INDEX `idx_name` (`name`),
  INDEX `idx_class` (`class_id`)
) COMMENT = '学生表';

#### 11、课程表

CREATE TABLE `course` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `course_code` VARCHAR(50) NOT NULL UNIQUE COMMENT '课程代码',
  `course_name` VARCHAR(100) NOT NULL COMMENT '课程名称',
  `credit` DECIMAL(3,1) NOT NULL COMMENT '学分',
  `total_hours` INT COMMENT '总学时',
  `theory_hours` INT COMMENT '理论学时',
  `practice_hours` INT COMMENT '实践学时',
  `course_type` TINYINT DEFAULT 1 COMMENT '课程类型：1-必修 2-选修 3-实践',
  `department_id` BIGINT COMMENT '开课院系',
  `description` TEXT COMMENT '课程描述',
  `assessment_method` VARCHAR(50) COMMENT '考核方式',
  `status` TINYINT DEFAULT 1 COMMENT '状态：1-开设 0-停开',
  `created_time` datetime DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  FOREIGN KEY (`department_id`) REFERENCES `department` (`id`),
  INDEX `idx_course_code` (`course_code`)
) COMMENT = '课程表';





#### 12、教学班表

CREATE TABLE `teaching_class` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `class_code` VARCHAR(50) NOT NULL UNIQUE COMMENT '教学班代码',
  `course_id` BIGINT NOT NULL COMMENT '课程ID',
  `teacher_id` BIGINT NOT NULL COMMENT '任课教师',
  `academic_year` VARCHAR(20) NOT NULL COMMENT '学年',
  `semester` TINYINT NOT NULL COMMENT '学期：1-春季 2-秋季',
  `max_students` INT DEFAULT 60 COMMENT '最大容量',
  `current_students` INT DEFAULT 0 COMMENT '当前选课人数',
  `class_time` VARCHAR(100) COMMENT '上课时间',
  `class_location` VARCHAR(100) COMMENT '上课地点',
  `status` TINYINT DEFAULT 1 COMMENT '状态：1-选课中 2-进行中 3-已结束',
  `created_time` datetime DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  FOREIGN KEY (`course_id`) REFERENCES `course` (`id`),
  FOREIGN KEY (`teacher_id`) REFERENCES `teacher` (`id`),
  INDEX `idx_year_semester` (`academic_year`, `semester`)
) COMMENT = '教学班表';





#### 13、选课表

CREATE TABLE `student_course` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `student_id` BIGINT NOT NULL,
  `teaching_class_id` BIGINT NOT NULL,
  `academic_year` VARCHAR(20) NOT NULL,
  `semester` TINYINT NOT NULL,
  `selected_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '选课时间',
  `usual_score` DECIMAL(5,2) COMMENT '平时成绩',
  `exam_score` DECIMAL(5,2) COMMENT '考试成绩',
  `total_score` DECIMAL(5,2) COMMENT '总评成绩',
  `grade_point` DECIMAL(3,2) COMMENT '绩点',
  `status` TINYINT DEFAULT 1 COMMENT '状态：1-在读 2-通过 3-未通过 4-退课',
  `created_time` datetime DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_student_teaching_class` (`student_id`, `teaching_class_id`),
  FOREIGN KEY (`student_id`) REFERENCES `student` (`id`),
  FOREIGN KEY (`teaching_class_id`) REFERENCES `teaching_class` (`id`),
  INDEX `idx_student_year_semester` (`student_id`, `academic_year`, `semester`)
) COMMENT = '选课表';





#### 14、专业课程关联表

CREATE TABLE `major_course` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `major_id` BIGINT NOT NULL COMMENT '专业ID',
  `course_id` BIGINT NOT NULL COMMENT '课程ID',
  `course_nature` TINYINT DEFAULT 1 COMMENT '课程性质：1-必修 2-选修 3-限选',
  `recommended_semester` INT COMMENT '建议修读学期',
  `created_time` datetime DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_major_course` (`major_id`, `course_id`),
  FOREIGN KEY (`major_id`) REFERENCES `major` (`id`) ON DELETE CASCADE,
  FOREIGN KEY (`course_id`) REFERENCES `course` (`id`) ON DELETE CASCADE
) COMMENT = '专业课程关联表';
